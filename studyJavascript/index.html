                    res.end('Hello, Node.js!\n');
                    });
                    
                    // 5. 서버가 3000번 포트에서 요청 대기 중임을 알립니다.
                    server.listen(3000, () => {
                    console.log('Server running at http://localhost:3000/');
                    });

Scope : 변수를 선언하고 변수가 적용되는 범위로, 변수가 어디까지 유효한지 범위를 가리키는 영역입니다.

Local Variable : 한 함수 안에서 사용할 수 있는 변수

Global Variable : 스크립트 소스 전체에서 사용할 수 있는 변수

1. var
var는 자바스크립트에서 오래된 변수 선언 방식으로, 전역 혹은 함수 범위로 지정됩니다. 

                // Global Scope(전역범위) 예시
                var globalVar = "I am a global variable";
                function checkGlobalVar() {
                console.log(globalVar); // 전역 변수에 접근 가능
                }

                checkGlobalVar(); // "I am a global variable"
                console.log(globalVar); // "I am a global variable"

                // Function Scope (함수 범위)예시
                function testVar() {
                var functionScopedVar = "I am inside a function";
                console.log(functionScopedVar); // 함수 내부에서는 접근 가능
                }

                testVar(); // "I am inside a function"
                console.log(functionScopedVar); // ReferenceError: functionScopedVar is not defined
1. var
그러나 같은 이름의 변수를 여러 번 선언할 수 있어 코드에 오류가 발생할 수 있으니 주의를 해야합니다. 

                var name = "data team";
                var name = "Aone";
                console.log(name);

2. let
let은 {} 블록 안에서만 유효하게 사용할 수 있는 선언방식입니다. 

                let greeting = "say Hi";
                let times = 4;

                if (times > 3) {
                        let hello = "say Hello instead";
                        console.log(hello);
                    }
                console.log(hello)
let은 재선언은 가능하나 범위 내에서 다시 선언을 할 순 없습니다. 

                // 재선언
                let greeting = "say Hi";
                greeting = "say Hello instead";

                // 같은 변수로 재선언
                let greeting = "say Hi";
                let greeting = "say Hello instead"; // error: Identifier 'greeting' has already been declared
3. const

const는 변수를 상수로 선언하기 때문에, 한 번 값을 할당하면 재할당할 수 없습니다.

따라서 변수 선언 시 즉시 초기화가 필요하며, 다음 두 가지 작업은 불가능합니다. 


                // 초기화를 하지 않고 선언하는 경우
                const age; // SyntaxError: Missing initializer in const declaration

                // 값을 재할당하는 경우
                const age = 30;
                age = 31; // TypeError: Assignment to constant variable.
3. const

만약 object(객체)로 사용하는 경우 해당 객체의 속성은 변경할 수 있습니다. 


                const person = {
                    name: "주영",
                    age: 27
                  };
                  
                  // 객체의 속성은 수정 가능
                  person.name = "재희";
                  person.age = 25;
                  
                  console.log(person);

3) 비교 연산자
두 값을 비교하는 데 사용됩니다. 결과는 true 또는 false입니다.

연산자 설명
== 값이 같은지 비교
=== 값과 타입이 모두 같은지 비교

3) 비교 연산자
두 값을 비교하는 데 사용됩니다. 결과는 true 또는 false입니다. 


                let x = 5;
                let y = "5";
                
                console.log(x == y); // true (값만 비교)
                console.log(x === y); // false (값과 타입을 모두 비교)
                
                console.log(x != y); // false (값만 비교)
                console.log(x !== y); // true (값과 타입을 모두 비교)
